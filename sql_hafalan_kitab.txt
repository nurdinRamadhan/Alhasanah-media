-- 2. Create Table hafalan_kitab
CREATE TABLE public.hafalan_kitab (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  santri_nis text not null,
  tanggal timestamp with time zone not null default now(),
  nama_kitab text not null, -- Jurumiah, Imrity, etc.
  bab_materi text null,     -- Contoh: Bab I'rab, Bab Kalam, dll.
  bait_awal integer null,   -- Untuk kitab Nadhom
  bait_akhir integer null,  -- Untuk kitab Nadhom
  halaman_awal integer null, -- Untuk kitab Natsar
  halaman_akhir integer null,
  predikat text null,       -- MUMTAZ, JAYYID, KURANG
  status text null,         -- LULUS, MENGULANG
  catatan text null,
  dicatat_oleh_id uuid null,
  
  constraint hafalan_kitab_pkey primary key (id),
  constraint hafalan_kitab_dicatat_oleh_id_fkey foreign KEY (dicatat_oleh_id) references profiles (id) on delete SET NULL,
  constraint hafalan_kitab_santri_nis_fkey foreign KEY (santri_nis) references santri (nis) on delete CASCADE
) TABLESPACE pg_default;

-- 3. Enable Row Level Security
ALTER TABLE public.hafalan_kitab ENABLE ROW LEVEL SECURITY;

-- 4. Create Policies (RLS)
-- Policy for Authenticated Users (Read All)
CREATE POLICY "Enable read access for all users" ON public.hafalan_kitab
  FOR SELECT USING (auth.role() = 'authenticated');

-- Policy for Admin/Ustadz (Insert/Update/Delete)
CREATE POLICY "Enable insert for authenticated users only" ON public.hafalan_kitab
  FOR INSERT WITH CHECK (auth.role() = 'authenticated');

CREATE POLICY "Enable update for users based on id" ON public.hafalan_kitab
  FOR UPDATE USING (auth.role() = 'authenticated');

CREATE POLICY "Enable delete for users based on id" ON public.hafalan_kitab
  FOR DELETE USING (auth.role() = 'authenticated');

-- 5. Create Index for Performance
CREATE INDEX idx_hafalan_kitab_santri ON public.hafalan_kitab(santri_nis);
CREATE INDEX idx_hafalan_kitab_tanggal ON public.hafalan_kitab(tanggal);
