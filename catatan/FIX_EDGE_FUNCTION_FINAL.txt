import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'
import admin from "npm:firebase-admin@11.10.1"

const serviceAccount = JSON.parse(Deno.env.get("FIREBASE_SERVICE_ACCOUNT_KEY")!);

if (!admin.apps.length) {
  admin.initializeApp({
    credential: admin.credential.cert(serviceAccount),
  });
}

Deno.serve(async (req) => {
  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  )

  try {
    // 1. Ambil data notifikasi SAJA (Tanpa JOIN ke user_devices)
    const { data: queue, error: fetchError } = await supabase
      .from('notification_queue')
      .select('id, user_id, title, body, data') // Spesifik kolom, tidak ada join disini
      .eq('status', 'pending')
      .limit(10);

    if (fetchError) throw fetchError;
    if (!queue || queue.length === 0) return new Response("No pending", { status: 200 });

    for (const item of queue) {
      // 2. Ambil token secara terpisah berdasarkan user_id
      const { data: devices } = await supabase
        .from('user_devices')
        .select('fcm_token')
        .eq('user_id', item.user_id);

      const tokens = devices?.map((d: any) => d.fcm_token) || [];
      
      if (tokens.length > 0) {
        try {
          await admin.messaging().sendMulticast({
            notification: { title: item.title, body: item.body },
            data: { 
                type: String(item.data?.type || 'manual'),
                id: String(item.id) 
            },
            tokens: tokens,
          });
          
          await supabase.from('notification_queue')
            .update({ status: 'sent', sent_at: new Date().toISOString() })
            .eq('id', item.id);
        } catch (e) {
          await supabase.from('notification_queue')
            .update({ status: 'failed', error_message: e.message })
            .eq('id', item.id);
        }
      } else {
        await supabase.from('notification_queue')
          .update({ status: 'failed', error_message: "No tokens found" })
          .eq('id', item.id);
      }
    }
    return new Response("Done", { status: 200 });
  } catch (err) {
    return new Response(err.message, { status: 500 });
  }
})
